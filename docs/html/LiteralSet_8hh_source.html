347template <typename ModelVec>
348inline void LiteralSet::discard_from_model(const ModelVec& model, int max_var,
349 vec<Lit>& discarded) {
350 discarded.clear();
351 if (_size == 0) return;
352
353 const int model_size = model.size();
354 const int limit = (max_var < model_size - 1) ? max_var : model_size - 1;
355
356 size_t removed = 0;
357
358 // Process in chunks for better cache utilization
359 for (int var = 1; var <= limit; var += 4) {
360 // Unroll loop manually for better performance
361 const int end = (var + 3 <= limit) ? var + 3 : limit;
362
363 for (int v = var; v <= end; ++v) {
364 const auto value = model[v];
365
366 // Branchless selection: compute both, use based on value
367 const size_t li_pos = literal_index(mkLit(v, false));
368 const size_t li_neg = literal_index(mkLit(v, true));
369
370 if (__builtin_expect(value == l_True, 1)) {
371 // Remove negative literal
372 if (__builtin_expect(li_neg < set.size() && set[li_neg], 1)) {
373 set[li_neg] = 0;
374 discarded.push(mkLit(v, true));
375 ++removed;
376 }
377 } else if (__builtin_expect(value == l_False, 1)) {
378 // Remove positive literal
379 if (__builtin_expect(li_pos < set.size() && set[li_pos], 1)) {
380 set[li_pos] = 0;
381 discarded.push(mkLit(v, false));
382 ++removed;
383 }
384 }
385 }
386
387 // Early exit if we've removed everything
388 if (__builtin_expect(removed >= _size, 0)) break;
389 }
390
391 _size -= removed;
392}
