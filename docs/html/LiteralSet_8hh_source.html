325template <typename ModelVec>
326inline void LiteralSet::discard_from_model(const ModelVec& model, int max_var,
327 vec<Lit>& discarded) {
328 discarded.clear();
329 if (_size == 0) return;
330
331 const int model_size = model.size();
332 const int limit = (max_var < model_size - 1) ? max_var : model_size - 1;
333
334 size_t removed = 0;
335
336 // Process in chunks for better cache utilization
337 for (int var = 1; var <= limit; var += 4) {
338 // Unroll loop manually for better performance
339 const int end = (var + 3 <= limit) ? var + 3 : limit;
340
341 for (int v = var; v <= end; ++v) {
342 const auto value = model[v];
343
344 // Branchless selection: compute both, use based on value
345 const size_t li_pos = literal_index(mkLit(v, false));
346 const size_t li_neg = literal_index(mkLit(v, true));
347
348 if (__builtin_expect(value == l_True, 1)) {
349 // Remove negative literal
350 if (__builtin_expect(li_neg < set.size() && set[li_neg], 1)) {
351 set[li_neg] = 0;
352 discarded.push(mkLit(v, true));
353 ++removed;
354 }
355 } else if (__builtin_expect(value == l_False, 1)) {
356 // Remove positive literal
357 if (__builtin_expect(li_pos < set.size() && set[li_pos], 1)) {
358 set[li_pos] = 0;
359 discarded.push(mkLit(v, false));
360 ++removed;
361 }
362 }
363 }
364
365 // Early exit if we've removed everything
366 if (__builtin_expect(removed >= _size, 0)) break;
367 }
368
369 _size -= removed;
370}
