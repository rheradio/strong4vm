23bool CheckCandidatesOneByOne::initialize() {
24 // Initialize the solver
25 solver.set_detector_weight(attention_weight);
26 for (Var i = 0; i <= max_id; ++i) {
27 solver.newVar();
28 }
29 vec<Lit> ls;
30 for (auto ci = clauses.begin(); ci != clauses.end(); ++ci) {
31 ls.clear();
32 for (auto li = (*ci).begin(); li != (*ci).end(); ++li) {
33 const Lit l = *li;
34 assert(var(l) <= max_id);
35 ls.push(l);
36 }
37 solver.addClause(ls);
38 }
39
40 // Get the first solution of the formula
41 const bool is_sat = solver.solve();
42
43 if (!is_sat) {
44 // Finish if the formula is unsatisfiable
45 return false;
46 }
47
48 // Initialize the candidates set with the literals of the first solution
49 const vec<lbool>& solution = solver.model;
50
51 for (Var variable = 1; variable <= max_id; ++variable) {
52 const lbool value = solution[variable];
53 if (value != l_Undef) {
54 const Lit l = mkLit(variable, value == l_False);
55 candidates.add(l);
56 }
57 }
58
59 // The formula is satisfiable. Let's go for the backbone!
60 return true;
61}

63// Run of the worker
64void CheckCandidatesOneByOne::run() {
65 while (candidates.size()) {
66 ++candidates_iterator;
67 const Lit candidate = *candidates_iterator;
68 vec<Lit> assumptions(1);
69 assumptions[0] = ~candidate;
70 const bool is_sat = bump_and_solve(assumptions);
71 if (!is_sat) {

80bool CheckCandidatesOneByOne::bump_and_solve(const vec<Lit>& assumptions) {
81 // Update the solver's detector activities and polarities
82 for (auto li = candidates.begin(); li != candidates.end(); ++li) {
83 const Lit l = *li;
84 const Var v = var(l);
85 solver.bump(v);
86 solver.set_detector_polarity(v, sign(l) ? l_False : l_True);
87 }
88 return solver.solve(assumptions);
