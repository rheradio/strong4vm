176 bool read_clause_count(const string& dimacs_path, int& clauses) {
177 ifstream dimacs_file(dimacs_path);
178 if (!dimacs_file.is_open()) {
179 error_message = "Could not open file: " + dimacs_path;
180 return false;
181 }
182
183 string line_str;
184 while (getline(dimacs_file, line_str)) {
185 if (line_str[0] == 'p') {
186 stringstream ss(line_str);
187 char p;
188 string cnf;
189 int vars;
190 ss >> p >> cnf >> vars >> clauses;
191 dimacs_file.close();
192 return true;
193 }
194 }
195 dimacs_file.close();
196 error_message = "No problem line found in DIMACS file";
197 return false;
198 }

267 void run() {
268 try {
269 // BackboneSolverAPI is pre-initialized, just process variables
270 // Process assigned variable range
271 for (int v = start_var; v <= end_var; v++) {
272 process_variable(v);
273 (*progress_counter)++;
274 }
275
276 success = true;
277 } catch (const exception& e) {
278 success = false;
279 error_msg = "Thread " + to_string(thread_id) +
280 " exception: " + e.what();
281 } catch (...) {
282 success = false;
283 error_msg = "Thread " + to_string(thread_id) +
284 ": Unknown exception";

299 void process_variable(int v) {
300 // Compute backbone assuming v=true
301 vector<int> assumptions = {v};
302 vector<int> line_vector =
303 bone_api->compute_backbone_with_assumptions(assumptions);
304
305 // Convert to indexed array for O(1) lookup
306 vector<int> line(num_variables + 1, 0);
307 for (int lit : line_vector) {
308 int var = abs(lit);
309 line[var] = lit;
310 }
311
312 // Extract requires edges
313 for (int i = 1; i <= num_variables; i++) {
314 if ((i != v) && (line[i] == i) && (global_bb[i] == 0)) {
315 requires_list << v << " " << i << "\n";
316 }
317 }
318
319 // Extract excludes edges
320 for (int i = v; i <= num_variables; i++) {
321 if ((line[i] == -i) && (global_bb[i] != -i) &&
322 (global_bb[v] != -v)) {
323 excludes_list << v << " " << i << "\n";
324 }
325 }
326 }

494 // Pre-create and initialize BackboneSolverAPI instances (single-threaded)

496 vector<unique_ptr<BackboneSolverAPI>> apis;
497 for (int t = 0; t < effective_threads; t++) {

556 // Check for errors (fail-fast)

561 return false;
