178 ifstream dimacs_file(dimacs_path);

213 bool read_feature_names(const string& dimacs_path,
214 map<int, string>& feature_map,
215 vector<bool>& aux_vars) {
216 ifstream dimacs_file(dimacs_path);
217 if (!dimacs_file.is_open()) {
218 error_message = "Could not open file: " + dimacs_path;
219 return false;
220 }
221
222 string line_str;
223 while (getline(dimacs_file, line_str)) {
224 if (line_str.empty()) continue;
225 if (line_str[0] == 'c') {
226 stringstream ss(line_str);
227 char c;
228 int var_number;
229 string word;
230
231 if (ss >> c >> var_number) {
232 stringstream full_name;
233 bool has_words = false;
234
235 while (ss >> word) {
236 has_words = true;
237 if (full_name.tellp() > 0) full_name << " ";
238 full_name << word;
239 }
240
241 if (has_words) {
242 string name = full_name.str();
243 feature_map[var_number] = name;
244
245 // Check if this is an auxiliary variable (starts with "aux_")
246 if (name.size() >= 4 && name.substr(0, 4) == "aux_") {
247 // Ensure aux_vars vector is large enough
248 if (var_number >= static_cast<int>(aux_vars.size())) {
249 aux_vars.resize(var_number + 1, false);
250 }
251 aux_vars[var_number] = true;
252 }
253 }
254 }
255 } else if (line_str[0] == 'p') {
256 // Problem line - we can stop after reading all comments
257 // (comments typically come before the problem line)
258 // But some formats have comments throughout, so continue
259 }
260 }
261 dimacs_file.close();
262 return true;
263 }

377 void process_variable(int v) {
378 // Compute backbone assuming v=true
379 vector<int> assumptions = {v};
380 vector<int> line_vector =
381 bone_api->compute_backbone_with_assumptions(assumptions);
382
383 // Convert to indexed array for O(1) lookup
384 vector<int> line(num_variables + 1, 0);
385 for (int lit : line_vector) {
386 int var = abs(lit);
387 line[var] = lit;
388 }
389
390 // Extract requires edges (skip edges to auxiliary variables)
391 for (int i = 1; i <= num_variables; i++) {
392 if ((i != v) && (line[i] == i) && (global_bb[i] == 0) && !is_aux(i)) {
393 requires_list << v << " " << i << "\n";
394 }
395 }
396
397 // Extract excludes edges (skip edges to auxiliary variables)
398 for (int i = v; i <= num_variables; i++) {
399 if ((line[i] == -i) && (global_bb[i] != -i) &&
400 (global_bb[v] != -v) && !is_aux(i)) {
401 excludes_list << v << " " << i << "\n";
402 }
403 }
404 }

609 excludes_list << v << " " << i << endl;

614 } else {

616 atomic<int> progress_counter(0);

618 // Pre-create and initialize BoneDiggerAPI instances (single-threaded)
619 cout << "Initializing " << effective_threads << " backbone solver instances..." << endl;
620 vector<unique_ptr<BoneDiggerAPI>> apis;
621 for (int t = 0; t < effective_threads; t++) {

637 workers.reserve(effective_threads); // Prevent reallocation
638 vector<thread> threads;
639 threads.reserve(effective_threads);
640
641 // Distribute vars_to_process indices across threads
642 int vars_per_thread = total_to_process / effective_threads;
643 int remainder = total_to_process % effective_threads;
644
645 int current_idx = 0;
646 for (int t = 0; t < effective_threads; t++) {
647 int start_idx = current_idx;
648 int count = vars_per_thread + (t < remainder ? 1 : 0);
649 int end_idx = start_idx + count - 1;

651 workers.emplace_back(ThreadWorker(

658 }

660 // Launch threads

663 }

675 thread.join();

678 cout << "\rProgress: " << total_to_process << " of "
679 << total_to_process << " variables" << endl;

683 if (!worker.success) {
684 error_message = worker.error_msg;
685 cerr << error_message << endl;
686 return false;
687 }

693 excludes_list << worker.excludes_list.str();
694 }
