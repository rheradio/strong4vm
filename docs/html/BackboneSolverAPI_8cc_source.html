247 bool read_file(const string& file_name) {
248 cleanup_detector();
249 cleanup_reader();
250
251 try {
252 // Open the file
253 gzFile ff = gzopen(file_name.c_str(), "rb");

255 return false;

257

265 clauses = dimacs_reader.get_clause_vector();
266 has_file = true;
267 is_sat = false;
268
269 return true;
270 } catch (...) {
271 cleanup_reader();
272 return false;
273 }

391 vector<int> compute() {
392 vector<int> result;
393
394 if (!has_file || detector_type == NONE) {
395 return result;
396 }
397
398 // Only cleanup if detector hasn't been created yet
399 if (detector == nullptr) {
400 try {
401 if (detector_type == ONE) {
402 CheckCandidatesOneByOne* one_detector = new CheckCandidatesOneByOne(max_id, clauses);
403 detector = one_detector;
404
405 if (!one_detector->initialize()) {
406 is_sat = false;
407 cleanup_detector();
408 return result;
409 }
410
411 is_sat = true;
412 one_detector->run();
413
414 // Extract backbone
415 for (Var v = 1; v <= max_id; ++v) {
416 if (one_detector->is_backbone(v)) {
417 int lit = one_detector->backbone_sign(v) ? v : -v;
418 result.push_back(lit);
419 }
420 }
421
422 } else if (detector_type == WITHOUT_ATTENTION) {
423 CheckCandidatesOneByOneWithoutAttention* without_detector =
424 new CheckCandidatesOneByOneWithoutAttention(max_id, clauses);
425 detector = without_detector;
426
427 if (!without_detector->initialize()) {
428 is_sat = false;
429 cleanup_detector();
430 return result;
431 }
432
433 is_sat = true;
434 without_detector->run();
435
436 // Extract backbone
437 for (Var v = 1; v <= max_id; ++v) {
438 if (without_detector->is_backbone(v)) {
439 int lit = without_detector->backbone_sign(v) ? v : -v;
440 result.push_back(lit);
441 }
442 }
443 }
444 } catch (...) {
445 cleanup_detector();
446 return vector<int>();
447 }
448 } else {
449 // Detector already exists, just extract backbone
450 try {
451 if (detector_type == ONE) {
452 CheckCandidatesOneByOne* one_detector = static_cast<CheckCandidatesOneByOne*>(detector);
453 for (Var v = 1; v <= max_id; ++v) {
454 if (one_detector->is_backbone(v)) {
455 int lit = one_detector->backbone_sign(v) ? v : -v;
456 result.push_back(lit);

458 }
459 } else if (detector_type == WITHOUT_ATTENTION) {
460 CheckCandidatesOneByOneWithoutAttention* without_detector =
461 static_cast<CheckCandidatesOneByOneWithoutAttention*>(detector);
462 for (Var v = 1; v <= max_id; ++v) {
463 if (without_detector->is_backbone(v)) {
464 int lit = without_detector->backbone_sign(v) ? v : -v;
465 result.push_back(lit);

468 }

555 vector<int> compute_with_assumps(const vector<int>& assumptions) {
556 vector<int> result;
557
558 if (!has_file || detector_type == NONE) {
559 return result;
560 }
561
562 // Convert int assumptions to MiniSat Lit assumptions
563 vec<Lit> minisat_assumptions;
564 for (int lit_int : assumptions) {
565 Var v = abs(lit_int);
566 bool sign = (lit_int < 0);
567 Lit l = mkLit(v, sign);
568 minisat_assumptions.push(l);
569 }
570
571 try {
572 if (detector_type == ONE) {
573 CheckCandidatesOneByOne* one_detector = new CheckCandidatesOneByOne(max_id, clauses);
574 bool is_sat_local = one_detector->run_with_assumptions(minisat_assumptions);
575
576 if (!is_sat_local) {
577 // Unsatisfiable under assumptions
578 delete one_detector;
579 return result;
580 }
581
582 // Extract backbone
583 for (Var v = 1; v <= max_id; ++v) {
584 if (one_detector->is_backbone(v)) {
585 int lit = one_detector->backbone_sign(v) ? v : -v;
586 result.push_back(lit);
587 }
588 }
589
590 delete one_detector;
591
592 } else if (detector_type == WITHOUT_ATTENTION) {
593 CheckCandidatesOneByOneWithoutAttention* without_detector =
594 new CheckCandidatesOneByOneWithoutAttention(max_id, clauses);
595 bool is_sat_local = without_detector->run_with_assumptions(minisat_assumptions);
596
597 if (!is_sat_local) {
598 // Unsatisfiable under assumptions
599 delete without_detector;
600 return result;
601 }
602
603 // Extract backbone
604 for (Var v = 1; v <= max_id; ++v) {
605 if (without_detector->is_backbone(v)) {
606 int lit = without_detector->backbone_sign(v) ? v : -v;
607 result.push_back(lit);
608 }
609 }
610
611 delete without_detector;
612 }
613 } catch (...) {
614 return vector<int>();
615 }
616
617 return result;
618 }

663 void print_bb() const {
664 if (!is_sat) {
665 cout << "Formula is UNSATISFIABLE" << endl;
666 return;
667 }
668
669 if (!has_file || detector_type == NONE) {
670 cout << "No backbone computed yet" << endl;
671 return;
672 }
673
674 // Print formula statistics
675 cout << "Formula statistics:" << endl;
676 cout << " Variables: " << max_id << endl;
677 cout << " Clauses: " << clauses.size() << endl;
678 cout << "Formula is SATISFIABLE" << endl;
679
680 // Count and collect backbone literals
681 vector<int> backbone_lits;
682
683 try {
684 if (detector_type == ONE) {
685 CheckCandidatesOneByOne* one_detector = static_cast<CheckCandidatesOneByOne*>(detector);
686 if (one_detector) {
687 for (Var v = 1; v <= max_id; ++v) {
688 if (one_detector->is_backbone(v)) {
689 int lit = one_detector->backbone_sign(v) ? v : -v;
690 backbone_lits.push_back(lit);
691 }
692 }
693 }
694 } else if (detector_type == WITHOUT_ATTENTION) {
695 CheckCandidatesOneByOneWithoutAttention* without_detector =
696 static_cast<CheckCandidatesOneByOneWithoutAttention*>(detector);
697 if (without_detector) {
698 for (Var v = 1; v <= max_id; ++v) {
699 if (without_detector->is_backbone(v)) {
700 int lit = without_detector->backbone_sign(v) ? v : -v;
701 backbone_lits.push_back(lit);
702 }
703 }
704 }
705 }
706 } catch (...) {
707 cout << "Error retrieving backbone" << endl;
708 return;
709 }
710
711 // Calculate percentage
712 double percentage = (max_id > 0) ? (100.0 * backbone_lits.size() / max_id) : 0.0;
713
714 cout << "Backbone size: " << backbone_lits.size() << " literals";
715 cout << " (" << std::fixed << std::setprecision(2) << percentage << "% of variables)" << endl;
716
717 if (backbone_lits.size() > 0) {
718 cout << "Backbone literals: ";
719 for (int lit : backbone_lits) {
720 cout << lit << " ";
721 }
722 cout << endl;
723 }
724 }
