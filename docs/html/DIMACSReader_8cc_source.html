116Lit DIMACSReader::parse_lit(Reader& in) {
117 Var v = 0;
118 bool neg = false;
119 if (*in == '-') {
120 neg = true;
121 ++in;
122 } else if (*in == '+')
123 ++in;
124 if ((*in < '0') || (*in > '9')) {
125 string s("unexpected char in place of a literal: ");
126 s += *in;
127 throw DIMACSReaderException(s);
128 }
129 while (*in >= '0' && *in <= '9') {
130 v = v * 10 + (*in - '0');
131 ++in;
132 }
133 return v ? mkLit(v, neg) : lit_Undef;
134}
