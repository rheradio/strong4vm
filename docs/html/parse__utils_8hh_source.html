163template<class B>
164static int parseInt(B& in) {
165 int val = 0;
166 bool neg = false;
167 skipWhitespace(in);
168 if (*in == '-') neg = true, ++in;
169 else if (*in == '+') ++in;
170 if (*in < '0' || *in > '9') fprintf(stderr, "PARSE ERROR! Unexpected char: %c\n", *in), exit(3);
171 while (*in >= '0' && *in <= '9')
172 val = val*10 + (*in - '0'),
173 ++in;
174 return neg ? -val : val; }
