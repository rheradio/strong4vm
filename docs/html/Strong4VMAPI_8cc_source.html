101 InputType detect_file_type(const std::string& filename) const {
102 fs::path path(filename);
103 std::string ext = path.extension().string();
104
105 // Convert to lowercase for case-insensitive comparison
106 std::transform(ext.begin(), ext.end(), ext.begin(),
107 [](unsigned char c) { return std::tolower(c); });
108
109 if (ext == ".uvl") {
110 return InputType::UVL;
111 } else if (ext == ".dimacs" || ext == ".cnf") {
112 return InputType::DIMACS;
113 }
114
115 return InputType::AUTO;

169 std::string validate_configuration(const AnalysisConfig& config) const {
170 // Check input file
171 if (config.input_file.empty()) {
172 return "Input file not specified";

176 return "Input file not found: " + config.input_file;

193

261 AnalysisResult perform_analysis(const AnalysisConfig& config) {
262 AnalysisResult result;
263 result.input_file = config.input_file;
264
265 // Validate configuration
266 std::string validation_error = validate_configuration(config);
267 if (!validation_error.empty()) {
268 result.success = false;
269 result.error_message = validation_error;
270 return result;
271 }
272
273 // Determine input type
274 InputType input_type = config.input_type;
275 if (input_type == InputType::AUTO) {
276 input_type = detect_file_type(config.input_file);
277 }
278 result.input_type = input_type;
279
280 // Determine output directory
281 std::string output_dir = config.output_dir;
282 if (output_dir.empty()) {
283 output_dir = get_directory(config.input_file);
284 }
285
286 // Ensure output directory exists
287 if (!fs::exists(output_dir)) {
288 try {
289 fs::create_directories(output_dir);
290 } catch (const std::exception& e) {
291 result.success = false;
292 result.error_message = "Failed to create output directory: " +
293 std::string(e.what());
294 return result;
295 }
296 }
297
298 std::string dimacs_file;
299 std::string basename = get_basename(config.input_file);
300 bool temp_dimacs = false;
301
302 // Step 1: Ensure we have a DIMACS file
303 if (input_type == InputType::UVL) {
304 if (verbose) {
305 std::cout << "=================================================\n";
306 std::cout << "Step 1: Converting UVL to DIMACS\n";
307 std::cout << "=================================================\n";
308 }
309
310 // Determine DIMACS output path
311 dimacs_file = output_dir + "/" + basename + ".dimacs";
312 temp_dimacs = !config.keep_dimacs;
313
314 // Convert UVL to DIMACS
315 uvl2dimacs::UVL2Dimacs converter;
316 converter.set_verbose(verbose);
317
318 // Set conversion mode
319 uvl2dimacs::ConversionMode uvl_mode =
320 (config.conversion_mode == ConversionMode::TSEITIN)
321 ? uvl2dimacs::ConversionMode::TSEITIN
322 : uvl2dimacs::ConversionMode::STRAIGHTFORWARD;
323
324 converter.set_mode(uvl_mode);
325
326 auto uvl_result = converter.convert(config.input_file, dimacs_file);
327
328 if (!uvl_result.success) {
329 result.success = false;
330 result.error_message = "UVL to DIMACS conversion failed: " +
331 uvl_result.error_message;
332 return result;
333 }
334
335 // Copy UVL statistics to result
336 result.num_features = uvl_result.num_features;
337 result.num_relations = uvl_result.num_relations;
338 result.num_constraints = uvl_result.num_constraints;
339 result.num_variables = uvl_result.num_variables;
340 result.num_clauses = uvl_result.num_clauses;
341
342 if (verbose) {
343 std::cout << "\nConversion successful!\n";
344 std::cout << " Features: " << uvl_result.num_features << "\n";
345 std::cout << " Variables: " << uvl_result.num_variables << "\n";
346 std::cout << " Clauses: " << uvl_result.num_clauses << "\n";
347 if (config.keep_dimacs) {
348 std::cout << " DIMACS file: " << dimacs_file << "\n";
349 }
350 std::cout << "\n";
351 }
352
353 if (config.keep_dimacs) {
354 result.dimacs_file = dimacs_file;
355 }
356
357 } else {
358 // Input is already DIMACS
359 dimacs_file = config.input_file;
360 basename = get_basename(dimacs_file);
361 }
362
363 // Step 2: Generate graphs from DIMACS
364 if (verbose) {
365 std::cout << "=================================================\n";
366 std::cout << "Step 2: Generating Strong Transitive Graphs\n";
367 std::cout << "=================================================\n";
368 }
369
370 dimacs2graphs::Dimacs2GraphsAPI graph_api;
371
372 std::string detector_str = detector_to_string(config.detector);
373 bool graph_success = graph_api.generate_graphs(
374 dimacs_file,
375 output_dir,
376 detector_str,
377 config.num_threads
378 );
379
380 if (!graph_success) {
381 result.success = false;
382 result.error_message = "Graph generation failed: " +
383 graph_api.get_error_message();
384
385 // Clean up temporary DIMACS file if created
386 if (temp_dimacs && fs::exists(dimacs_file)) {
387 fs::remove(dimacs_file);
388 }
389
390 return result;
391 }
392
393 // Update statistics if we didn't have them from UVL conversion
394 if (input_type == InputType::DIMACS) {
395 result.num_variables = graph_api.get_num_variables();
396 result.num_clauses = graph_api.get_num_clauses();
397 }
398
399 // Get global backbone
400 result.global_backbone = graph_api.get_global_backbone();
401
402 // Separate core and dead features from backbone
403 for (int lit : result.global_backbone) {
404 if (lit > 0) {
405 result.core_features.push_back(lit);
406 } else {
407 result.dead_features.push_back(-lit); // Store as positive variable
408 }
409 }
410
411 // Set output file paths
412 result.requires_graph_file = output_dir + "/" + basename + "__requires.net";
413 result.excludes_graph_file = output_dir + "/" + basename + "__excludes.net";
414 result.core_features_file = output_dir + "/" + basename + "__core.txt";
415 result.dead_features_file = output_dir + "/" + basename + "__dead.txt";
416
417 if (verbose) {
418 std::cout << "\nGraph generation successful!\n";
419 std::cout << " Variables: " << result.num_variables << "\n";
420 std::cout << " Clauses: " << result.num_clauses << "\n";
421 std::cout << " Core features: " << result.core_features.size() << "\n";
422 std::cout << " Dead features: " << result.dead_features.size() << "\n";
423 std::cout << "\nOutput files:\n";
424 std::cout << " " << result.requires_graph_file << "\n";
425 std::cout << " " << result.excludes_graph_file << "\n";
426 std::cout << " " << result.core_features_file << "\n";
427 std::cout << " " << result.dead_features_file << "\n";
428
429 if (verbose) {
430 std::cout << "\n=================================================\n";
431 std::cout << "Analysis Complete!\n";
432 std::cout << "=================================================\n";
433 }
434 }
435
436 // Clean up temporary DIMACS file if needed
437 if (temp_dimacs && fs::exists(dimacs_file)) {
438 fs::remove(dimacs_file);
439 }
440
441 result.success = true;
442 return result;
443 }
